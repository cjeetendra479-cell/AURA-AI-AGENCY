import tensorflow as tf import numpy as np # 1. DATA: Maan lo input 'x' hai aur output 'y' (y = 2x - 1 ka logic hai) x_train = np.array([-1.0, 0.0, 1.0, 2.0, 3.0, 4.0], dtype=float) y_train = np.array([-3.0, -1.0, 1.0, 3.0, 5.0, 7.0], dtype=float) # 2. MODEL DESIGN: Ek simple layer wala Neural Net model = tf.keras.Sequential([ # Input layer se seedha output layer (1 neuron) tf.keras.layers.Dense(units=1, input_shape=[1]) ]) # 3. COMPILE: Optimizer (galti sudharne wala) aur Loss (galti check karne wala) model.compile(optimizer='sgd', loss='mean_squared_error') # 4. TRAINING: Model ko 500 baar data dikhao (Epochs) print("Training Neural Network...") model.fit(x_train, y_train, epochs=500, verbose=0) # 5. PREDICTION: Ab naye data par test karo print("Prediction for 10.0 is:", model.predict([10.0])) # Result 19.0 ke bahut paas aayega (kyunki 2*10 - 1 = 19)
